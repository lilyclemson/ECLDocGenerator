\BLOCK{ macro threetag(name, tuples) }
\BLOCK{ if tuples|length > 0 }
\par
\begin{description}
\BLOCK{ for t in tuples }
\BLOCK{ if name == "Parameter" }
\item [\colorbox{parametertag}{\color{white} \textbf{\textsf{\VAR{name.upper()}}}}] \textbf{\underline{\VAR{t[0]|escape_tex}}} ||| \VAR{t[1]|escape_tex} --- \VAR{t[2]|escape_tex}
\BLOCK{ else }
\item [\colorbox{tagtype}{\color{white} \textbf{\textsf{\VAR{name.upper()}}}}] \textbf{\underline{\VAR{t[0]|escape_tex}}} ||| \VAR{t[1]|escape_tex} --- \VAR{t[2]|escape_tex}
\BLOCK{ endif }
\BLOCK{ endfor }
\end{description}
\BLOCK{ endif }
\BLOCK{ endmacro }

\BLOCK{ macro seetag(name, tuples, libNames) }
\BLOCK{ if tuples|length > 0 }
\par
\begin{description}
\BLOCK{ for t in tuples }
\BLOCK{ if t[0].split('.')[0] in libNames }
\item [\colorbox{tagtype}{\color{white} \textbf{\textsf{\VAR{name.upper()}}}}]\href{https://cdn.hpccsystems.com/pdf/ml/\VAR{t[0].split('.')[0]}.pdf}{\VAR{t[0].split('.')[0]}}.\VAR{'.'.join(t[0].split('.')[1: ])}
\BLOCK{ else }
\item [\colorbox{tagtype}{\color{white} \textbf{\textsf{\VAR{name.upper()}}}}] \hyperlink{ecldoc:\VAR{t[0].lower()}}{\VAR{t[0] | escape_tex}}
\BLOCK{ endif }
\BLOCK{ endfor }
\end{description}
\BLOCK{ endif }
\BLOCK{ endmacro }
